# 二叉树的按层打印与ZigZag打印
## 题目
给定一颗二叉树的头结点head,分别实现按成打印和ZigZag打印二叉树的函数。
## 按层打印的实现
### C++实现:
```cpp
struct Node{
	int value;
	Node* left;
	Node* right;
	Node(int x) : value(x), left(NULL), right(NULL) {}
};

void printByLevel(Node* head) {
	if (NULL == head)
		return;
	queue<Node*> mq;
	Node* last = head; // 每层最右边的结点
	int level = 1;

	mq.push(head); // 根结点先入队
	cout << "Level " << level << ": ";
	while (!mq.empty()) {
		Node* cur = mq.front();
		cout << cur->value << " ";
		// 把当前结点的孩子结点入队
		if (NULL != cur->left)
			mq.push(cur->left);
		if (NULL != cur->right)
			mq.push(cur->right);

		if (cur == last) {
			last = mq.back(); // 下一层的最后一个结点
			++level;
			cout << endl;
			if (cur != last)
				cout << "Level " << level << ": ";
		}

		// 当前结点出队
		mq.pop();
	}
}
```

## 按ZigZag打印的实现
### C++实现:
#### 方法1
```cpp
struct Node {
	int value;
	Node* left;
	Node* right;
	Node(int x) : value(x), left(NULL), right(NULL) {}
};

void printByZigZag(Node* head) {
	if (NULL == head)
		return;
	queue<Node*> mq;	// 按层入队
	stack<Node*> ms;	// 保存下一层的全部结点
	int level = 1;
	mq.push(head);

	while(!mq.empty()) {
		Node* cur = NULL;
		// 奇数层，从左往右打印
		// 偶数层，从右往左打印
		if (1 == level % 2) {
			cout << "Level " << level << "from left to right: ";
			// 把下一层从左往右入栈
			while (!mq.empty) {
				cur = mq.front();
				if (NULL != cur->left)
					ms.push(cur->left);
				if (NULL != cur->right)
					ms.push(cur->right);

				// 打印结点值
				cout << cur->value << " ";
				mq.pop();
			}
		} else {
			cout << "Level " << level << "from right to left: ";
			// 下一层结点从右往左入栈
			while (!mq.empty) {
				cur = mq.front();
				if (NULL != cur->right)
					ms.push(cur->right);
				if (NULL != cur->left)
					ms.push(cur->left);
				cout << cur->value << " ";
				mq.pop();
			}
		}

		cout << endl;
		
		// 把下一层结点入队
		while (!ms.empty()) {
			mq.push(ms.top());
			ms.pop();
		}
		
		++level;
	}
}
```
#### 方法2
使用双端队列实现。创建一个双端队列dq，每层结点入队出队按以下两个原则：

    原则1：从左到右遍历时，队头结点出队，该结点的孩子结点按照从左到右的顺序从队尾入队，当然，孩子结点为空，不用入队。
    
    原则2：从右往左遍历时，队尾结点出队，该结点的孩子结点按照从右到左的顺序从队头入队，同样，孩子结点为空，不用入队。
    
这两个原则的使用时机在于如何确定每一层最后一个结点，用last表示每层最后一个结点

    从原则1切换到原则2：遍历到当前层最后一个结点时，下一层的last = dq.front();
    
    从原则2切换到原则1：last = dq.back();
```cpp
struct Node {
	int value;
	Node* left;
	Node* right;
	Node(int x) : value(x), left(NULL), right(NULL) {}
};

void printByZigZag(Node* head) {
	if (NULL == head) 
		return;
	deque<Node*> dq;
	Node* last = head;
	int level = 1;
	dq.push_front(head);
	cout << "Level " << level << " from left to right: ";

	while (!dq.empty()) {
		// 从左往右遍历
		// 1)从队头出队
		// 2)从队尾入队:先左孩子，后右孩子
		if (1 == level % 2) {
			head = dq.front();
			dq.pop_front();
			cout << head->value << " ";
			if (NULL != head->left) 
				dq.push_back(head->left);
			if (NULL != head->right)
				dq.push_back(head->right);

			// 遍历到当前层最右结点
			if (head == last && !dq.empty()) {
				// 下一层最后遍历的结点在最左边
				// 也就是队列的队头
				last = dq.front();
				++level;
				cout << endl;
				cout << "Level " << level << " from right to left: ";
			}
		} else {
			// 从右往左遍历
			// 1)从队尾出队
			// 2)从队头入队:先右孩子,后左孩子
			head = dq.back();
			dq.pop_back();
			cout << head->value << " ";
			if (NULL != head->right)
				dq.push_front(head->right);
			if (NULL != head->left)
				dq.push_front(head->left);

			if (head == last && !dq.empty()) {
				last = dq.back();
				++level;
				cout << endl;
				cout << "Level " << level << " from left to right: ";
			}
		}
		cout << endl;
	}
```

# 遍历二叉树的神级方法

## 题目

给定一颗二叉树的头结点head，完成二叉树的先序、中序和后续遍历。如果二叉树的结点数为N，要求时间复杂度为O(N)，额外的空间复杂度为O(1)。

## 思路

为了满足空间复杂度为O(1)的限制，可以使用Morris遍历。Morris遍历的实质就是避免使用栈结构，而是让树的下层结点指向上层结点，可以让下层结点指向null的空闲指针指向上层的某个结点，这样就可以从下层回到上层。

## 中序遍历

### 遍历过程

1. 假设当前子树的头结点为`head`，获取`head`的左子树`head->left`。
    1）左子树不为空，把左子树中的最右结点的`right`指针指向`head`，然后head的左子树继续步骤1；
    2）左子树为空，把`head`记为`node`，进入步骤2。

2. 从`node`开始通过每个结点的`right`指针进行移动，并依次打印。对于每个移动到的结点`cur`都判断该结点的左子树的最右结点是否指向该结点`cur`。
    1）如果是。让`cur`结点的左子树的最右结点的right指针指向空，这将把树的结构调整回来，然后打印`cur`，继续通过`cur`的`right`指针移动到下一个结点，重复步骤2。
    2）如果不是。以`cur`为头的子树重回步骤1。

### C++

```cpp
struct Node {
	int value;
	Node* left;
	Node* right;
	Node(int x) : value(x), left(NULL), right(NULL) {}
};


// 中序遍历
void morrisInOrder(Node* head) {
	if (NULL == head)
		return;
	Node* cur1 = head;
	Node* cur2 = NULL;
	while (NULL != cur1) {
		cur2 = cur1->left;  // 当前子树的左子树
		if (NULL != cur2) {
		    // 左子树不空，找到其最右结点
			while (NULL != cur2->right && cur2->right != cur1) {
				cur2 = cur2->right;
			}
			// 按情况对最右结点进行操作
			if (cur2->right == NULL) {
				cur2->right = cur1;
				cur1 = cur1->left;
				continue;
			} else {
				cur2->right = NULL;
			}
		}
		
		cout << cur1->value << " ";
		cur1 = cur1->right;
	}	
	cout << endl;
}
```

## 先序遍历

与Morris中序遍历就是打印时间的不同，将先序遍历的打印时机放在步骤1中每次找子树的时候。

### C++

```cpp
// 先序遍历
void morrisPreOrder(Node* head) {
	if (NULL == head)
		return;
	Node* cur1 = head;
	Node* cur2 = NULL;
	while (NULL != cur1) {
		cur2 = cur1->left; // 左子树根结点
		if (NULL != cur2) {
			// 左子树不空，找到左子数的最右结点
			while (NULL != cur2->right && cur2->right != cur1) {
				cur2 = cur2->right;
			}
			if (NULL == cur2->right) {
				cur2->right = cur1;
				cout << cur1->value << " ";
				cur1 = cur1->left;
				continue;
			} else {
				cur2->right = NULL;
			}
		} else {
			cout << cur1->value << " ";
		}
		cur1 = cur1->right;
	}
	cout << endl;
}
```

## 后续遍历

也是中序遍历的改写。依次逆序打印所有结点的左子树的右边界，在每次将树结构调回去的时候打印。

### C++

```cpp
// 后序遍历
// 反转右边界
Node* reverseEdge(Node* head) {
	Node *pre = NULL;
	Node *next = NULL;
	while (NULL != head) {
		next = head->right;
		head->right = pre;
		pre = head;
		head = next;
	}
	return pre;
}

// 打印右边界
void printEdge(Node *head) {
	Node* tail = reverseEdge(head);
	Node* cur = tail;
	while (NULL != cur) {
		cout << cur->value << " ";
		cur = cur->right;
	} 
	reverseEdge(tail);
}

void morrisPosOrder(Node* head) {
	if (NULL == head)
		return;
	Node *cur1 = head;
	Node *cur2 = NULL;

	while (NULL != cur1) {
		cur2 = cur1->left;
		if (NULL != cur2) {
			while (NULL != cur2->right && cur2->right != cur1) {
				cur2 = cur2->right;
			}
			if (NULL == cur2->right) {
				cur2->right = cur1;
				cur1 = cur1->left;
				continue;
			} else {
				cur2->right = NULL;
				printEdge(cur1->left);
			}
		}
		cur1 = cur1->right;
	}
	// 打印根结点所在的右边界
	printEdge(head);
	cout << endl;
}
```
